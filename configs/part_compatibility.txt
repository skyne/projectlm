# Assembly compatibility rules (slot-based + matrix)
# rule= starts a new rule block

# Carbon brakes need a carbon-class chassis
rule=ceramic_requires_carbon_chassis
if_slot=brake_system
if_part=CarbonCeramic
requires_slot=chassis
requires_any_parts=LMHInHouse,LMHDallaraBuilt,LMHMultimaticBuilt,LMHMonocoque,LMDhDallara,LMDhOreca,LMDhMultimatic,LMDhLigier,Oreca07,GT3Spaceframe,GT3Oreca,GT3PrattMiller,GT3McLaren,GT3Multimatic

rule=brembo_requires_carbon_chassis
if_slot=brake_system
if_part=BremboHypercar
requires_slot=chassis
requires_any_parts=LMHInHouse,LMHDallaraBuilt,LMHMultimaticBuilt,LMHMonocoque,LMDhDallara,LMDhOreca,LMDhMultimatic,LMDhLigier

rule=akebono_requires_lmh_chassis
if_slot=brake_system
if_part=AkebonoHypercar
requires_slot=chassis
requires_any_parts=LMHInHouse,LMHMonocoque,LMHDallaraBuilt

rule=apr_requires_prototype_chassis
if_slot=brake_system
if_part=APRacingPrototype
requires_slot=chassis
requires_any_parts=LMHInHouse,LMHMonocoque,LMDhOreca,LMDhDallara,LMDhMultimatic,LMDhLigier

# Le Mans low-drag rear floor needs matching low-drag nose
rule=wingless_requires_low_drag_nose
if_slot=rear_aero
if_part=WinglessGroundEffect
requires_slot=front_aero
requires_any_parts=LowDragNose,LowDragNoseSlim

# APRacing GT3 kit fits homologated GT3 constructors
rule=apr_gt3_requires_gt3_chassis
if_slot=brake_system
if_part=APRacingGT3
requires_slot=chassis
requires_any_parts=GT3Spaceframe,GT3Oreca,GT3PrattMiller,GT3McLaren,GT3Multimatic

# LMDh spec hybrid fits any homologated LMDh spine
rule=lmdh_hybrid_requires_lmdh_spine
if_slot=hybrid_system
if_part=LMDh50kW
requires_slot=chassis
requires_any_parts=LMDhDallara,LMDhOreca,LMDhMultimatic,LMDhLigier
